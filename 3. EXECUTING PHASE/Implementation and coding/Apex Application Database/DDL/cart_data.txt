CREATE TABLE  "CARTLIST" 
   (	"CARTLIST_ID" NUMBER(5,0), 
	"QUANTITY" NUMBER(5,0) NOT NULL ENABLE, 
	"PRODUCT_ID" NUMBER(5,0), 
	"USER_ID" NUMBER(5,0), 
	 PRIMARY KEY ("CARTLIST_ID") ENABLE
   )
/
ALTER TABLE  "CARTLIST" ADD FOREIGN KEY ("PRODUCT_ID")
	  REFERENCES  "PRODUCT" ("PRODUCT_ID") ENABLE
/
ALTER TABLE  "CARTLIST" ADD FOREIGN KEY ("USER_ID")
	  REFERENCES  "USERS" ("USER_ID") ENABLE
/

CREATE OR REPLACE TRIGGER  "CARTLIST_TRIGGER" 
BEFORE INSERT ON CARTLIST 
FOR EACH ROW 
BEGIN 
    IF :NEW.cartlist_id IS NULL THEN 
        :NEW.cartlist_id := cartlist_seq.NEXTVAL; 
    END IF; 
END; 

/
ALTER TRIGGER  "CARTLIST_TRIGGER" ENABLE
/
